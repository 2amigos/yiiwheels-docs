105     /**
106      * Helper method to display different input types for the different complain bootstrap forms wrapped with their
107      * labels, help and error messages. This method is a replacement of the old 'typeRow' methods from Yii-Bootstrap
108      * extension. Example:
109      * <pre>
110      *     $form->row(WhHtml::INPUT_TEXT, $model, 'attribute', array('style'=>'width:125px'));
111      *    $form->row(WhHtml::INPUT_DROPDOWN, $model, 'attribute', array('a'=>'A','b'=>'B'), array());
112      * </pre>
113      * @param $type
114      * @param $model
115      * @param $attribute
116      * @param $data
117      * @param array $htmlOptions
118      * @return string
119      * @throws CException
120      */
121     public function row($type, $model, $attribute, $data = array(), $htmlOptions = array())
122     {
123         if (!in_array($type, WhHtml::validInputs()))
124             throw new CException(Yii::t('tb', 'Unrecognized input type'));
125 
126         $labelOptions = WhHtml::popOption('labelOptions', $htmlOptions, array());
127         $errorOptions = WhHtml::popOption('errorOptions', $htmlOptions, array());
128         $containerOptions = WhHtml::popOption('containerOptions', $htmlOptions, array());
129 
130         $labelOptions = WhHtml::defaultOption('formType', $this->type, $labelOptions);
131 
132         ob_start();
133 
134         // make sure it holds the class control-label
135         if ($this->type === WhHtml::FORM_HORIZONTAL)
136             echo CHtml::openTag('div', WhHtml::addClassName('control-group', $containerOptions));
137 
138         // form's inline do not render labels and radio|checkbox input types render label's differently
139         if ($this->type !== WhHtml::FORM_INLINE
140             && !preg_match('/radio|checkbox/i', $type)
141             && WhHtml::popOption('label', $htmlOptions, true)
142         )
143             echo CHtml::activeLabel($model, $attribute, $labelOptions);
144         elseif (preg_match('/radio|checkbox/i', $type))
145             $htmlOptions['labelOptions'] = $labelOptions;
146 
147         if (WhHtml::popOption('block', $htmlOptions, false))
148             $htmlOptions = WhHtml::addClassName('input-block-level', $htmlOptions);
149 
150         $params = in_array($type, WhHtml::$dataInputs)
151             ? array($model, $attribute, $data, $htmlOptions)
152             : array($model, $attribute, $htmlOptions);
153 
154         $errorSpan = $this->error($model, $attribute, $errorOptions);
155 
156         echo $this->wrapControl(call_user_func_array('WhHtml::active' . ucfirst($type), $params), $errorSpan); /* since PHP 5.3 */
157 
158         if ($this->type == WhHtml::FORM_VERTICAL && WhHtml::popOption('error', $htmlOptions, true))
159             echo $errorSpan;
160 
161         if ($this->type == WhHtml::FORM_HORIZONTAL)
162             echo '</div>';
163 
164         return ob_get_clean();
165 
166     }
