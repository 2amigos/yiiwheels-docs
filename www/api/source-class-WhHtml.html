307     /**
308      * Extracts widget configuration options from array of htmlOptions
309      * @param string $widget the widget name
310      * @param array $htmlOptions additional HTML options
311      * @param array $options the default widget options
312      * @return array
313      */
314     protected static function getWidgetConfigurationOptions($widget, $htmlOptions, $options = array())
315     {
316         $configurationOptions = self::$configurationOptions[$widget];
317 
318         $options = self::moveOptions(
319             $configurationOptions,
320             $htmlOptions,
321             $options);
322 
323         $options['htmlOptions'] = self::removeOptions($htmlOptions, $configurationOptions);
324 
325         return $options;
326     }
