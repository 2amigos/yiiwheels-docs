 18     /**
 19      * @var string $url the route to call via AJAX to get the data from
 20      */
 21     public $url;

 23     /**
 24      * @var string $cssClass the class name that will wrap up the cell content.
 25      * Important Note: this class will be used as the trigger for the AJAX call, so make sure is unique for the
 26      * column.
 27      */
 28     public $cssClass = 'wh-relational-column';
 29 
 30     /**
 31      * @var bool $cacheData if set to true, there won't be more than one AJAX request. If set to false, the widget will
 32      * continuously make AJAX requests. This is useful if the data could vary. If the data doesn't change then is better
 33      * to set it to true. Defaults to true.
 34      */
 35     public $cacheData = true;
 36 
 37     /**
 38      * @var string a javascript function that will be invoked if an AJAX call occurs.
 39      *
 40      * The function signature is <code>function(tr, rowid, data)</code>
 41      * <ul>
 42      * <li><code>tr</code> is the newly created TR HTML object that will display the returned server data.</li>
 43      * <li><code>rowid</code> the model id of the row.</li>
 44      * <li><code>data</code> is the data returned by the server that is already displayed on the row.</li>
 45      * </ul>
 46      * Note: This handler is not called for JSONP requests.
 47      *
 48      * Example (add in a call to TbRelationalColumn):
 49      * <pre>
 50      *  ...
 51      *  'afterAjaxUpdate'=>'js:function(tr,rowid, data){ console.log(rowid); }',
 52      *  ...
 53      * </pre>
 54      */
 55     public $afterAjaxUpdate;
 56 
 57     /**
 58      * @var string $ajaxErrorMessage the message that is displayed on the newly created row in case there is an AJAX
 59      * error.

 61     public $ajaxErrorMessage = 'Error';

 76     /**
 77      * Overrides CDataColumn renderDataCell in order to wrap up its content with the object that will be used as a
 78      * trigger.
 79      * Important: Making use of links as a content for this of column is an error.
 80      * @param int $row
 81      */
 82     public function renderDataCell($row)
 83     {
 84         $data    = $this->grid->dataProvider->data[$row];
 85         $options = $this->htmlOptions;
 86         if ($this->cssClassExpression !== null) {
 87             $class = $this->evaluateExpression($this->cssClassExpression, array('row' => $row, 'data' => $data));
 88             if (isset($options['class']))
 89                 $options['class'] .= ' ' . $class;
 90             else
 91                 $options['class'] = $class;
 92         }
 93         echo CHtml::openTag('td', $options);
 94         echo CHtml::openTag('span', array('class' => $this->cssClass, 'data-rowid' => $this->getPrimaryKey($data)));
 95         $this->renderDataCellContent($row, $data);
 96         echo '</span>';
 97         echo '</td>';
 98     }

100     /**
101      * Helper function to return the primary key of the $data
102      *  * IMPORTANT: composite keys on CActiveDataProviders will return the keys joined by comma
103      * @param CActiveRecord $data
104      * @return null|string
105      */
106     protected function getPrimaryKey($data)
107     {
108         if ($this->grid->dataProvider instanceof CActiveDataProvider) {
109             $key = $this->grid->dataProvider->keyAttribute === null ? $data->getPrimaryKey(
110             ) : $data->{$this->keyAttribute};
111             return is_array($key) ? implode(',', $key) : $key;
112         }
113         if ($this->grid->dataProvider instanceof CArrayDataProvider || $this->grid->dataProvider instanceof CSqlDataProvider)
114             return is_object(
115                 $data
116             ) ? $data->{$this->grid->dataProvider->keyField} : $data[$this->grid->dataProvider->keyField];
117 
118         return null;
119     }

121     /**
122      * Register script that will handle its behavior
123      */
124     public function registerClientScript()
125     {
126         Yii::app()->bootstrap->registerAssetCss('bootstrap-relational.css');
127         /** @var $cs CClientScript */
128         $cs = Yii::app()->getClientScript();
129         if ($this->afterAjaxUpdate !== null) {
130             if ((!$this->afterAjaxUpdate instanceof CJavaScriptExpression) && strpos(
131                     $this->afterAjaxUpdate,
132                     'js:'
133                 ) !== 0
134             )
135                 $this->afterAjaxUpdate = new CJavaScriptExpression($this->afterAjaxUpdate);
136         } else
137             $this->afterAjaxUpdate = 'js:$.noop';
138 
139         $this->ajaxErrorMessage = CHtml::encode($this->ajaxErrorMessage);
140         $afterAjaxUpdate        = CJavaScript::encode($this->afterAjaxUpdate);
141         $span                   = count($this->grid->columns);
142         $loadingPic             = CHtml::image(Yii::app()->yiiwheels->getAssetsUrl() . '/img/loading.gif');
143         $cache                  = $this->cacheData ? 'true' : 'false';
144         $data                   = !empty($this->submitData) && is_array(
145             $this->submitData
146         ) ? $this->submitData : 'js:{}';
147         $data                   = CJavascript::encode($data);
148 
149         $js = <<<EOD
150 $(document).on('click','.{$this->cssClass}', function(){
151     var span = $span;
152     var that = $(this);
153     var status = that.data('status');
154     var rowid = that.data('rowid');
155     var tr = $('#relatedinfo'+rowid);
156     var parent = that.parents('tr').eq(0);
157     var afterAjaxUpdate = {$afterAjaxUpdate};
158 
159     if (status && status=='on'){return}
160     that.data('status','on');
161 
162     if (tr.length && !tr.is(':visible') && {$cache})
163     {
164         tr.slideDown();
165         that.data('status','off');
166         return;
167     }else if (tr.length && tr.is(':visible'))
168     {
169         tr.slideUp();
170         that.data('status','off');
171         return;
172     }
173     if (tr.length)
174     {
175         tr.find('td').html('{$loadingPic}');
176         if (!tr.is(':visible')){
177             tr.slideDown();
178         }
179     }
180     else
181     {
182         var td = $('<td/>').html('{$loadingPic}').attr({'colspan':$span});
183         tr = $('<tr/>').prop({'id':'relatedinfo'+rowid}).append(td);
184         /* we need to maintain zebra styles :) */
185         var fake = $('<tr class="hide"/>').append($('<td/>').attr({'colspan':$span}));
186         parent.after(tr);
187         tr.after(fake);
188     }
189     var data = $.extend({$data}, {id:rowid});
190     $.ajax({
191         url: '{$this->url}',
192         data: data,
193         success: function(data){
194             tr.find('td').html(data);
195             that.data('status','off');
196             if ($.isFunction(afterAjaxUpdate))
197             {
198                 afterAjaxUpdate(tr,rowid,data);
199             }
200         },
201         error: function()
202         {
203             tr.find('td').html('{$this->ajaxErrorMessage}');
204             that.data('status','off');
205         }
206     });
207 });
208 EOD;
209         $cs->registerScript(__CLASS__ . '#' . $this->id, $js);
210     }
